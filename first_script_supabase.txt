create table if not exists type(
  id_type serial primary key,
  type text
);

create table if not exists location(
  id_location serial primary key,
  location text
);

create table if not exists product(
  id_product serial primary key,
  product text,
  quantity integer,
  id_type integer references type(id_type),
  id_location integer references location(id_location)
);

create table if not exists role(
  id_role serial primary key,
  role text
);

--el tipo uuid se utiliza para que sea el mismo tipo de la tabla auth.users
create table if not exists users(
  id_user uuid primary key references auth.users (id) on DELETE CASCADE,
  email text,
  username text,
  name text,
  id_role integer references role(id_role),
  first_sign_in boolean
);

create table if not exists type_register(
  id_type_register serial primary key,
  type_register text
);

create table if not exists register(
  id_register serial primary key,
  id_type_register integer references type_register(id_type_register),
  id_product integer references product(id_product),
  id_user uuid references users(id_user),
  date timestamp,
  quantity integer
);
