import { serve } from "https://deno.land/std@0.168.0/http/server.ts"
import { createClient } from "https://esm.sh/@supabase/supabase-js@2"

serve(async (req) => {
  const { email, current_password, new_password, id_user } = await req.json()

  const supabaseUrl = Deno.env.get('SUPABASE_URL') ?? ''
  const anonKey = Deno.env.get('SUPABASE_ANON_KEY') ?? '' 
  const serviceRoleKey = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY') ?? ''


  const userClient = createClient(supabaseUrl, anonKey)


  const adminClient = createClient(supabaseUrl, serviceRoleKey)

  try {
  
    const { error: loginError } = await userClient.auth.signInWithPassword({
      email,
      password: current_password,
    })

    if (loginError) {
      return new Response(JSON.stringify({ error: "La contraseña actual no es válida" }), {
        status: 401,
        headers: { "Content-Type": "application/json" },
      })
    }

    
    const { error: authError } = await adminClient.auth.admin.updateUserById(
      id_user,
      { password: new_password }
    )
    if (authError) throw authError

   
    const { error: dbError } = await adminClient
      .from('users')
      .update({ first_sign_in: false })
      .eq('id_user', id_user)

    if (dbError) throw dbError

    return new Response(JSON.stringify({ message: "Contraseña actualizada" }), {
      status: 200,
      headers: { "Content-Type": "application/json" },
    })

  } catch (error) {
    return new Response(JSON.stringify({ error: error.message }), {
      status: 400,
      headers: { "Content-Type": "application/json" },
    })
  }
})
